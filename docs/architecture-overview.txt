    %% Gateway flow
    AIDL --> Queue
    Voice --> Queue
    Notif --> Queue
    Cron --> Queue

    %% Approval flow
    Queue --> Whitelist
    Whitelist --> Sensitive
    Sensitive --> Severity
    Severity --> Policy
    Policy -->|ask| Confirm
    Confirm --> E2B
    Policy -->|auto| E2B

    %% Cognitive flow
    E2B -->|simple task| FC
    E2B -->|complex task| E4B
    E4B --> FC
    Context --> E4B
    Loader --> Context
    Registry --> Loader
    BuiltIn --> Registry
    Custom --> Registry
    RoomDB --> Context
    CmdHistory --> RoomDB
    ConvCtx --> RoomDB
    AppPattern --> RoomDB
    ContactCtx --> RoomDB

    %% Execution flow
    FC --> Router
    Router --> T1
    Router --> T2
    Router --> T3
    Router --> T4
    T3 --> PathA
    T3 --> PathB

    %% Verification
    T1 --> Verify
    T2 --> Verify
    T3 --> Verify
    T4 --> Verify
    Verify -->|success| SM
    Verify -->|fail| ErrorRec
    ErrorRec -->|retry| Router
    ErrorRec -->|abort| SM

    %% State
    SM --> Kill
    SM -.->|update progress| AIDL
